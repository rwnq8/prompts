```python
def calculate_gap_requirements(performance_specs, definitions):
    """Calculate band gap requirements from performance specs"""
    
    temp_k = performance_specs.get('Operating_Temperature', {}).get('quantified_value_K')
    user_gap = performance_specs.get('Topological_Gap', {}).get('quantified_value_meV')
    
    requirements = {}
    
    if user_gap:
        requirements['target_gap_meV'] = user_gap
        requirements['source'] = 'user_specified'
    elif temp_k:
        # Calculate minimum gap from temperature: E_g > 3.5*k_B*T for stability
        k_b = 0.08617  # meV/K
        min_gap = 3.5 * k_b * temp_k
        requirements['target_gap_meV'] = min_gap
        requirements['source'] = 'calculated_from_temperature'
        requirements['calculation'] = f'E_g_min = 3.5 * k_B * T = 3.5 * 0.08617 * {temp_k}'
    
    if 'target_gap_meV' in requirements:
        requirements['definition'] = definitions.get('Performance_Terms', {}).get(
            'topological_gap', 'Energy difference protecting edge states')
    
    return requirements
```

```python
def extract_growth_constraints(fabrication_constraints, definitions):
    """Extract growth method constraints"""
    
    growth_methods = fabrication_constraints.get('Growth_Methods', {})
    user_prefs = growth_methods.get('user_preferences', [])
    compatibility = growth_methods.get('compatibility_analysis', {})
    
    constraints = []
    
    for method in user_prefs:
        if method in compatibility:
            method_info = compatibility[method]
        else:
            method_info = {}
        
        constraints.append({
            'method': method,
            'definition': definitions.get('Fabrication_Terms', {}).get(method, 
                        f'Growth method: {method}'),
            'temperature_range': method_info.get('temperature_range', 'to_be_determined'),
            'precision': method_info.get('precision_level', 'typical_for_method')
        })
    
    return constraints
```
