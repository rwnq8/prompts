class PureLinearAlgebra:
    """Pure Python linear algebra implementation"""
    
    @staticmethod
    def matrix_multiply(A, B):
        """Matrix multiplication without numpy"""
        n = len(A)
        m = len(B[0])
        p = len(B)
        result = [[0+0j for _ in range(m)] for _ in range(n)]
        
        for i in range(n):
            for j in range(m):
                for k in range(p):
                    result[i][j] += A[i][k] * B[k][j]
        return result
    
    @staticmethod
    def matrix_add(A, B):
        """Matrix addition"""
        n = len(A)
        m = len(A[0])
        result = [[0+0j for _ in range(m)] for _ in range(n)]
        
        for i in range(n):
            for j in range(m):
                result[i][j] = A[i][j] + B[i][j]
        return result
    
    @staticmethod
    def jacobi_eigensolver(A, max_iter=100, tol=1e-10):
        """Jacobi method for symmetric/Hermitian matrices"""
        n = len(A)
        V = [[1.0 if i == j else 0.0 for j in range(n)] for i in range(n)]
        D = [row[:] for row in A]  # Copy
        
        for _ in range(max_iter):
            # Find largest off-diagonal element
            max_val = 0
            p, q = 0, 0
            for i in range(n):
                for j in range(i+1, n):
                    if abs(D[i][j]) > max_val:
                        max_val = abs(D[i][j])
                        p, q = i, j
            
            if max_val < tol:
                break
            
            # Compute rotation
            if abs(D[p][p] - D[q][q]) < 1e-14:
                theta = math.pi / 4
            else:
                theta = 0.5 * math.atan2(2*D[p][q].real, D[p][p].real - D[q][q].real)
            
            c = math.cos(theta)
            s = math.sin(theta)
            
            # Apply rotation to D
            for i in range(n):
                if i not in (p, q):
                    dip = D[i][p]
                    diq = D[i][q]
                    D[i][p] = c * dip - s * diq
                    D[p][i] = D[i][p].conjugate()
                    D[i][q] = s * dip + c * diq
                    D[q][i] = D[i][q].conjugate()
            
            dp = D[p][p]
            dq = D[q][q]
            dpq = D[p][q]
            D[p][p] = c*c*dp + s*s*dq - 2*c*s*dpq.real
            D[q][q] = s*s*dp + c*c*dq + 2*c*s*dpq.real
            D[p][q] = D[q][p] = 0+0j
            
            # Update eigenvectors
            for i in range(n):
                vip = V[i][p]
                viq = V[i][q]
                V[i][p] = c * vip - s * viq
                V[i][q] = s * vip + c * viq
        
        eigenvalues = [D[i][i].real for i in range(n)]
        return eigenvalues, V

class TopologicalPhysicsEngine:
    """Physics calculations for topological materials"""
    
    def __init__(self, physics_params):
        self.physics_params = physics_params
        
        # Extract Hamiltonian parameters
        self.hamiltonian_type = physics_params.get('hamiltonian_type', 'BHZ')
        self.coefficients = physics_params.get('coefficients', {})
        self.disorder_params = physics_params.get('disorder_params', {})
    
    def build_hamiltonian(self, kx, ky):
        """Build Hamiltonian at k-point"""
        if self.hamiltonian_type == 'BHZ':
            return self._build_bhz_hamiltonian(kx, ky)
        elif self.hamiltonian_type == 'Kane-Mele':
            return self._build_kane_mele_hamiltonian(kx, ky)
        else:
            return self._build_generic_hamiltonian(kx, ky)
    
    def _build_bhz_hamiltonian(self, kx, ky):
        """Build Bernevig-Hughes-Zhang Hamiltonian (4x4)"""
        m0 = self.coefficients.get('m0', 0.0)
        A1 = self.coefficients.get('A1', 0.0)
        A2 = self.coefficients.get('A2', 0.0)
        B1 = self.coefficients.get('B1', 0.0)
        B2 = self.coefficients.get('B2', 0.0)
        C = self.coefficients.get('C', 0.0)
        D1 = self.coefficients.get('D1', 0.0)
        D2 = self.coefficients.get('D2', 0.0)
        
        k2 = kx*kx + ky*ky
        
        # Construct 4x4 Hamiltonian
        H = [[0+0j for _ in range(4)] for _ in range(4)]
        
        # Diagonal elements
        epsilon_k = C - D1 * k2
        H[0][0] = epsilon_k + (m0 - B1 * k2)
        H[1][1] = epsilon_k + (m0 - B1 * k2)
        H[2][2] = epsilon_k - (m0 - B2 * k2)
        H[3][3] = epsilon_k - (m0 - B2 * k2)
        
        # Off-diagonal elements
        H[0][3] = A1 * kx - 1j * A2 * ky
        H[1][2] = A1 * kx + 1j * A2 * ky
        H[2][1] = A1 * kx - 1j * A2 * ky
        H[3][0] = A1 * kx + 1j * A2 * ky
        
        # Make Hermitian
        for i in range(4):
            for j in range(i):
                H[j][i] = H[i][j].conjugate()
        
        return H
    
    def add_disorder(self, H, disorder_strength=None):
        """Add disorder to Hamiltonian"""
        if disorder_strength is None:
            disorder_strength = self.disorder_params.get('strength', 0.0)
        
        n = len(H)
        disorder_H = [[0+0j for _ in range(n)] for _ in range(n)]
        
        disorder_type = self.disorder_params.get('type', 'random_potential')
        
        if disorder_type == 'random_potential':
            for i in range(n):
                disorder_H[i][i] = random.uniform(-1, 1) * disorder_strength
        
        # Add disorder to Hamiltonian
        total_H = PureLinearAlgebra.matrix_add(H, disorder_H)
        return total_H
    
    def calculate_topological_gap(self):
        """Calculate minimum topological gap"""
        # Sample k-points in Brillouin zone
        n_samples = 20
        k_points = []
        for i in range(n_samples):
            for j in range(n_samples):
                kx = (i/n_samples - 0.5) * 2 * math.pi
                ky = (j/n_samples - 0.5) * 2 * math.pi
                k_points.append((kx, ky))
        
        min_gap = float('inf')
        
        for kx, ky in k_points:
            H = self.build_hamiltonian(kx, ky)
            H = self.add_disorder(H)
            
            eigenvalues, _ = PureLinearAlgebra.jacobi_eigensolver(H)
            eigenvalues.sort()
            
            # Find gap between valence and conduction bands
            if len(eigenvalues) >= 4:  # For 4-band model
                gap = eigenvalues[2] - eigenvalues[1]  # Gap between middle two bands
                if gap < min_gap:
                    min_gap = gap
        
        # Convert to meV (assume eV units in Hamiltonian)
        return min_gap * 1000

        min_gap_target = recipe_params.get('target_specs', {}).get('min_gap_meV', 50)
        if gap >= min_gap_target:
            working_samples += 1
    
    # Device works if majority of realizations pass
    device_works = (working_samples / n_disorder_samples) >= 0.8
    
    return {
        'growth_result': growth_result,
        'mean_gap_meV': sum(gaps) / len(gaps) if gaps else 0,
        'min_gap_meV': min(gaps) if gaps else 0,
        'working_samples': working_samples,
        'total_samples': n_disorder_samples,
        'device_works': device_works
    }

def run_monte_carlo_simulation(recipe_params, n_fabrication_runs, n_disorder_samples):
    """Run Monte Carlo simulation"""
    
    working_devices = 0
    detailed_results = []
    
    for run in range(n_fabrication_runs):
        run_result = simulate_fabrication_run(recipe_params, n_disorder_samples)
        detailed_results.append(run_result)
        
        if run_result['device_works']:
            working_devices += 1
    
    predicted_yield = working_devices / n_fabrication_runs if n_fabrication_runs > 0 else 0
    
    return predicted_yield, detailed_results
