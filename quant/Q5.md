def analyze_process_windows(recipe, simulation_results):
    """Analyze process windows using tolerances and simulation results"""
    
    critical_params = recipe['process_control'].get('critical_parameters', [])
    
    analysis = []
    overall_cpk = 1.0
    
    for param in critical_params:
        target = param.get('target_value')
        tolerance = param.get('tolerance')
        control_method = param.get('control_method', 'typical')
        
        if target and tolerance:
            # Estimate typical control capability (σ) based on control method
            typical_sigma = estimate_typical_sigma(control_method)
            
            # Calculate Cpk
            cpk = calculate_cpk(target, tolerance, typical_sigma)
            
            analysis.append({
                'parameter': param.get('parameter'),
                'target': target,
                'tolerance': tolerance,
                'typical_control': typical_sigma,
                'cpk': cpk,
                'assessment': assess_cpk(cpk)
            })
            
            # Track worst-case Cpk
            overall_cpk = min(overall_cpk, cpk)
    
    return {
        'score': min(100, overall_cpk * 33.33),  # Scale: Cpk=3 → 100%
        'critical_parameters': analysis,
        'overall_process_capability': f"Cpk = {overall_cpk:.2f}",
        'assessment': 'CAPABLE' if overall_cpk >= 1.33 else 'MARGINAL' if overall_cpk >= 1.0 else 'INCORPABLE'

def predict_manufacturing_yield(simulation_results, equipment_compatibility, process_windows):
    """Predict manufacturing yield by combining simulation and manufacturing factors"""
    
    simulation_yield = simulation_results.get('predicted_yield', 0.5)
    equipment_reliability = equipment_compatibility.get('score', 70) / 100 * 0.95  # Scale and apply typical reliability
    process_capability = process_windows.get('score', 70) / 100
    
    # Operator dependence based on process complexity
    process_steps = count_process_steps(recipe)
    operator_dependence = calculate_operator_dependence(process_steps)
    
    # Calculate total predicted yield
    predicted_yield = simulation_yield * equipment_reliability * process_capability * operator_dependence
    
    # Calculate confidence interval
    simulation_std = simulation_results.get('gap_std_meV', 0) / simulation_results.get('mean_gap_meV', 1) * 0.1
    yield_variance = predicted_yield * simulation_std
    confidence_interval = [max(0, predicted_yield - 1.96 * yield_variance), min(1, predicted_yield + 1.96 * yield_variance)]
    
    return {
        'predicted_yield': predicted_yield,
        'confidence_interval': confidence_interval,
        'components': {
            'simulation_yield': simulation_yield,
            'equipment_reliability': equipment_reliability,
            'process_capability': process_capability,
            'operator_dependence': operator_dependence
        },
        'economic_assessment': {
            'minimum_viable_yield': 0.3,  # Typical for research prototypes
            'commercial_viability': predicted_yield > 0.7,
            'assessment': 'VIABLE_FOR_RESEARCH' if predicted_yield >= 0.3 else 'NOT_VIABLE'
        }
    }

def analyze_failure_modes(recipe, simulation_results):
    """Analyze potential failure modes using physics and manufacturing principles"""

    # Growth-related failures
    growth_failures = identify_growth_failures(recipe)

        probability = estimate_failure_probability(failure, simulation_results)

            'rpn': rpn,

        probability = estimate_failure_probability(failure, simulation_results)
        impact = assess_failure_impact(failure)
        detection = assess_detection_capability(failure, recipe)
        
        rpn = probability * impact * (1 - detection)

            'probability': probability,
            'impact': impact,
            'detection': detection,
            'rpn': rpn,
            'mitigation': suggest_mitigation(failure)
        })
    
    # Operational failures
    operational_failures = identify_operational_failures(recipe)
    for failure in operational_failures:
        probability = estimate_failure_probability(failure, simulation_results)
        impact = assess_failure_impact(failure)
        detection = assess_detection_capability(failure, recipe)
        
        rpn = probability * impact * (1 - detection)

            'category': 'operational',
            'probability': probability,
            'impact': impact,
            'detection': detection,
            'rpn': rpn,
            'mitigation': suggest_mitigation(failure)
        })
    
    # Sort by RPN
    failure_modes.sort(key=lambda x: x['rpn'], reverse=True)
    
    # Calculate overall risk score
    total_rpn = sum(fm['rpn'] for fm in failure_modes)
    avg_rpn = total_rpn / len(failure_modes) if failure_modes else 0
    
    return {
        'score': min(100, 100 - avg_rpn * 10),  # Convert RPN to score
        'top_failure_modes': failure_modes[:5],  # Top 5
        'total_failure_modes': len(failure_modes),
        'assessment': 'LOW_RISK' if avg_rpn < 20 else 'MODERATE_RISK' if avg_rpn < 50 else 'HIGH_RISK'
    }

        'failure_risk': assessments['failure'].get('score', 0)
    }

def generate_feedback_vectors(decision, assessments, recipe):
    """Generate feedback vectors based on decision and assessments"""
    
    feedback = {
        'required': decision == 'REOPTIMIZE_FABRICATION',
        'primary_issue': identify_primary_issue(assessments),
        'root_causes': identify_root_causes(assessments),
        'recommended_improvements': []
    }
    
    if decision == 'REOPTIMIZE_FABRICATION':
        # Generate specific improvements
        improvements = []
        
        # Process window improvements
        for param in assessments['process'].get('critical_parameters', []):
            if param.get('cpk', 0) < 1.33:
                improvements.append({
                    'improvement': f"Widen tolerance or improve control for {param['parameter']}",
                    'expected_impact': f"Cpk improvement from {param['cpk']:.2f} to target 1.33",
                    'yield_improvement': estimate_yield_improvement(param['cpk'], 1.33),
                    'cost_impact': 'Low to moderate',
                    'priority': 'HIGH' if param['cpk'] < 1.0 else 'MEDIUM'
                })
        
        # Equipment improvements
        for concern in assessments['equipment'].get('concerns', []):
            improvements.append({
                'improvement': f"Address equipment concern: {concern}",
                'expected_impact': 'Improved equipment compatibility',
                'yield_improvement': 0.05,  # Estimated 5% yield improvement
                'cost_impact': 'Equipment dependent',
                'priority': 'MEDIUM'
            })
        
        feedback['recommended_improvements'] = improvements
        
        # Calculate total expected yield after improvements
        current_yield = assessments['yield'].get('predicted_yield', 0)
        total_improvement = sum(imp.get('yield_improvement', 0) for imp in improvements)
        feedback['total_expected_yield_after_improvements'] = min(1.0, current_yield + total_improvement)
    
    return feedback
